LV2 오픈 채팅방

1차시도 (성공)
def solution(record):
    
    user = {r.split(" ")[1]:r.split(" ")[2] for r in record if len(r.split(" ")) == 3}

    answer = []
    for r in record:
        code = r.split(" ")
        op = code[0]
        id = code[1]

        if op == "Enter":
            answer.append(user[id]+"님이 들어왔습니다.")
        elif op == "Leave":
            answer.append(user[id]+"님이 나갔습니다.")
    
    return answer

dict형태로 UID를 키로 닉넴임을 값으로 저장하도록 하고 입장이나 닉네임 변경이나 명령이 3항이기 때문에
3항인 것들을 구분하여 저장하도록 하며 입장과 변경이 순차적으로 진행되기때문에 dict에서 키에해당하는 값을
저장하는 형태는 입장과 변경 모두 적용이 가능하다고 판단했다.
